// Helper: check if user has a specific permission (from role or user-specific override)
async function hasPermission(userId, permissionName) {
    if (!userId || !permissionName) return false;
    
    const [rows] = await db.query(
        `
        SELECT 
            COALESCE(up.is_granted, rp.permission_id) as has_perm
        FROM permissions p
        LEFT JOIN role_permissions rp ON p.permission_id = rp.permission_id
        LEFT JOIN user_roles ur ON rp.role_id = ur.role_id AND ur.user_id = ?
        LEFT JOIN user_permissions up ON p.permission_id = up.permission_id AND up.user_id = ?
        WHERE p.permission_name = ?
        LIMIT 1
        `,
        [userId, userId, permissionName]
    );
    
    return rows.length > 0 && rows[0].has_perm !== null && rows[0].has_perm !== 0;
}
